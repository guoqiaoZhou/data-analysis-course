```python
import numpy as np
import pandas as pd

np.random.seed(42)

def generate_base_data(n=1000):
    """生成基础模拟数据模板"""
    # 协变量
    age = np.random.normal(35, 10, n).astype(int)
    age = np.clip(age, 18, 65)
    
    gender = np.random.choice([0, 1], n, p=[0.45, 0.55])
    
    # 连续变量（对数正态分布模拟收入）
    income = np.random.lognormal(8, 0.8, n)
    
    # 处理分配（带选择偏差）
    treatment_prob = 0.3 + 0.4 * (income > np.median(income))
    treatment_prob = np.clip(treatment_prob, 0.05, 0.95)
    treatment = np.random.binomial(1, treatment_prob)
    
    # 结果变量
    base_outcome = 50 + 0.1 * income + 5 * treatment
    outcome = base_outcome + np.random.normal(0, 10, n)
    outcome = np.maximum(outcome, 0)
    
    df = pd.DataFrame({
        'user_id': range(1, n + 1),
        'age': age,
        'gender': gender,
        'income': np.round(income, 2),
        'treatment': treatment,
        'outcome': np.round(outcome, 2)
    })
    return df
```

```python
def generate_panel_data(n_units=200, n_periods=10, treatment_period=5, true_effect=3.0):
    """生成DID面板数据"""
    units = np.arange(n_units)
    periods = np.arange(n_periods)
    
    # 单位固定效应
    unit_fe = np.random.normal(0, 2, n_units)
    
    # 时间固定效应（线性趋势）
    time_fe = np.linspace(0, 1, n_periods)
    
    # 处理组分配（前50%为处理组）
    is_treated = (units < n_units // 2).astype(int)
    
    data = []
    for i in units:
        for t in periods:
            treated = is_treated[i] * (t >= treatment_period)
            y = (unit_fe[i] + time_fe[t] + true_effect * treated 
                 + np.random.normal(0, 1))
            data.append({
                'unit_id': i,
                'period': t,
                'post': int(t >= treatment_period),
                'treated': is_treated[i],
                'did': treated,
                'y': y
            })
    return pd.DataFrame(data)
```

```python
from sklearn.linear_model import LogisticRegression
from scipy.spatial.distance import cdist

def estimate_propensity(df, covariates, treatment_col='treatment'):
    """估计倾向得分"""
    X = df[covariates]
    T = df[treatment_col]
    
    model = LogisticRegression(max_iter=1000)
    model.fit(X, T)
    
    df['propensity_score'] = model.predict_proba(X)[:, 1]
    return df

def nearest_neighbor_matching(df, caliper=None, treatment_col='treatment'):
    """最近邻匹配（1:1）"""
    treated = df[df[treatment_col] == 1].copy()
    control = df[df[treatment_col] == 0].copy()
    
    # 计算距离矩阵
    distances = cdist(treated[['propensity_score']], 
                      control[['propensity_score']], metric='euclidean')
    
    matched_pairs = []
    used_control = set()
    
    for i, treat_idx in enumerate(treated.index):
        min_dist_idx = np.argmin(distances[i])
        min_dist = distances[i, min_dist_idx]
        
        if caliper is not None and min_dist > caliper:
            continue
        if min_dist_idx in used_control:
            continue
            
        used_control.add(min_dist_idx)
        control_idx = control.index[min_dist_idx]
        matched_pairs.append((treat_idx, control_idx))
    
    return matched_pairs
```

### 2.2 平衡性检验（SMD）

```python
def compute_smd(df, var, treatment_col='treatment'):
    """计算单个变量的标准化均值差（SMD）"""
    treated = df[df[treatment_col] == 1][var]
    control = df[df[treatment_col] == 0][var]
    
    mean_diff = treated.mean() - control.mean()
    pooled_std = np.sqrt((treated.var() + control.var()) / 2)
    
    smd = mean_diff / pooled_std if pooled_std > 0 else 0
    return smd

def balance_table(df, vars_list, treatment_col='treatment'):
    """生成平衡性检验表"""
    results = []
    for var in vars_list:
        smd = compute_smd(df, var, treatment_col)
        results.append({'variable': var, 'SMD': round(smd, 3), 
                        'balanced': abs(smd) < 0.1})
    return pd.DataFrame(results)
```

### 2.4 DML估计（交叉验证）

```python
from sklearn.linear_model import LassoCV, RidgeCV
from sklearn.ensemble import GradientBoostingRegressor
from sklearn.model_selection import KFold

def dml_estimator(df, treatment_col='D', outcome_col='Y', 
                  covariate_cols=None, model_type='lasso', n_splits=5):
    """双重机器学习估计器"""
    if covariate_cols is None:
        covariate_cols = [c for c in df.columns if c.startswith('X')]
    
    X = df[covariate_cols].values
    D = df[treatment_col].values
    Y = df[outcome_col].values
    
    kf = KFold(n_splits=n_splits, shuffle=True, random_state=42)
    
    theta_estimates = []
    
    for train_idx, test_idx in kf.split(X):
        X_train, X_test = X[train_idx], X[test_idx]
        D_train, D_test = D[train_idx], D[test_idx]
        Y_train, Y_test = Y[train_idx], Y[test_idx]
        
        # 估计nuisance函数
        if model_type == 'lasso':
            g_model = LassoCV(random_state=42)
            m_model = LassoCV(random_state=42)
        else:
            g_model = GradientBoostingRegressor(random_state=42)
            m_model = GradientBoostingRegressor(random_state=42)
        
        g_model.fit(X_train, Y_train)
        m_model.fit(X_train, D_train)
        
        # 在测试集上预测
        g_pred = g_model.predict(X_test)
        m_pred = m_model.predict(X_test)
        
        # Neyman正交化得分
        residual_Y = Y_test - g_pred
        residual_D = D_test - m_pred
        
        theta = np.mean(residual_Y * residual_D) / np.mean(residual_D ** 2)
        theta_estimates.append(theta)
    
    return np.mean(theta_estimates), np.std(theta_estimates)
```

### 3.1 t检验全家桶

```python
from scipy import stats

def t_test_summary(group_a, group_b, paired=False, alpha=0.05):
    """t检验汇总：自动选择合适的方法"""
    if paired:
        # 配对t检验
        stat, pvalue = stats.ttest_rel(group_a, group_b)
        test_name = "配对t检验"
    else:
        # 先检验方差齐性（Levene检验）
        _, levene_p = stats.levene(group_a, group_b)
        
        if levene_p < 0.05:
            # 方差不齐 → Welch's t-test
            stat, pvalue = stats.ttest_ind(group_a, group_b, equal_var=False)
            test_name = "Welch's t检验"
        else:
            # 方差齐性 → Student's t-test
            stat, pvalue = stats.ttest_ind(group_a, group_b, equal_var=True)
            test_name = "Student's t检验"
    
    # 效应量（Cohen's d）
    pooled_std = np.sqrt((np.var(group_a, ddof=1) + np.var(group_b, ddof=1)) / 2)
    cohens_d = (np.mean(group_a) - np.mean(group_b)) / pooled_std
    
    # 置信区间
    mean_diff = np.mean(group_a) - np.mean(group_b)
    se = pooled_std * np.sqrt(1/len(group_a) + 1/len(group_b))
    ci_low = mean_diff - stats.t.ppf(1-alpha/2, len(group_a)+len(group_b)-2) * se
    ci_high = mean_diff + stats.t.ppf(1-alpha/2, len(group_a)+len(group_b)-2) * se
    
    return {
        'test': test_name,
        'statistic': stat,
        'pvalue': pvalue,
        'cohens_d': cohens_d,
        'mean_diff': mean_diff,
        'ci_95': (ci_low, ci_high),
        'significant': pvalue < alpha
    }
```

### 3.2 Bootstrap置信区间

```python
def bootstrap_ci(data, statistic_func=np.mean, n_bootstrap=5000, ci=0.95):
    """Bootstrap置信区间"""
    n = len(data)
    boot_stats = []
    
    for _ in range(n_bootstrap):
        sample = np.random.choice(data, size=n, replace=True)
        boot_stats.append(statistic_func(sample))
    
    boot_stats = np.array(boot_stats)
    alpha = (1 - ci) / 2
    ci_lower = np.percentile(boot_stats, alpha * 100)
    ci_upper = np.percentile(boot_stats, (1 - alpha) * 100)
    
    return {
        'estimate': statistic_func(data),
        'ci_lower': ci_lower,
        'ci_upper': ci_upper,
        'boot_std': np.std(boot_stats),
        'boot_stats': boot_stats
    }
```

### 3.3 样本量计算

```python
def sample_size_per_group(delta, sigma, alpha=0.05, power=0.8):
    """计算每组所需样本量"""
    from scipy.stats import norm
    
    z_alpha = norm.ppf(1 - alpha / 2)
    z_beta = norm.ppf(power)
    
    n = 2 * ((z_alpha + z_beta) * sigma / delta) ** 2
    return int(np.ceil(n))

def mde_from_sample_size(n, sigma, alpha=0.05, power=0.8):
    """已知样本量，计算可检测的最小效应"""
    from scipy.stats import norm
    
    z_alpha = norm.ppf(1 - alpha / 2)
    z_beta = norm.ppf(power)
    
    delta = (z_alpha + z_beta) * sigma * np.sqrt(2 / n)
    return delta
```

### 4.1 CUPED方差缩减

```python
def cuped_adjust(y, x, theta=None):
    """CUPED方差缩减调整"""
    if theta is None:
        theta = np.cov(y, x)[0, 1] / np.var(x)
    
    y_cuped = y - theta * (x - np.mean(x))
    return y_cuped

def cuped_variance_reduction(y, x):
    """计算CUPED方差缩减比例"""
    y_cuped = cuped_adjust(y, x)
    var_original = np.var(y, ddof=1)
    var_cuped = np.var(y_cuped, ddof=1)
    reduction = (var_original - var_cuped) / var_original
    return reduction
```

```python
def simulate_aa_test(n_per_group=5000, n_simulations=1000, alpha=0.05):
    """模拟AA检验：检验假阳性率"""
    false_positives = 0
    pvalues = []
    
    for _ in range(n_simulations):
        # 两组来自同一分布
        group_a = np.random.normal(0, 1, n_per_group)
        group_b = np.random.normal(0, 1, n_per_group)
        
        _, pvalue = stats.ttest_ind(group_a, group_b)
        pvalues.append(pvalue)
        
        if pvalue < alpha:
            false_positives += 1
    
    fpr = false_positives / n_simulations
    return {
        'false_positive_rate': fpr,
        'expected_fpr': alpha,
        'pvalues': np.array(pvalues)
    }
```

### 4.3 多重检验校正（BH法）

```python
def benjamini_hochberg(pvalues, alpha=0.05):
    """Benjamini-Hochberg FDR控制"""
    pvalues = np.array(pvalues)
    n = len(pvalues)
    
    sorted_idx = np.argsort(pvalues)
    sorted_p = pvalues[sorted_idx]
    
    # 找到最大的k使得 p_(k) <= alpha * k / n
    thresholds = alpha * np.arange(1, n + 1) / n
    rejections = sorted_p <= thresholds
    
    if np.any(rejections):
        max_k = np.max(np.where(rejections)[0])
        rejected = sorted_idx[:max_k + 1]
    else:
        rejected = np.array([])
    
    return rejected
```

### 5.3 WOE-IV分箱

```python
from sklearn.tree import DecisionTreeClassifier

def tree_based_bins(x, y, n_bins=10):
    """基于决策树的最优分箱"""
    x = x.reshape(-1, 1)
    
    # 用决策树找到最优分割点
    tree = DecisionTreeClassifier(max_leaf_nodes=n_bins, random_state=42)
    tree.fit(x, y)
    
    # 提取分割阈值
    tree_thresholds = tree.tree_.threshold
    tree_thresholds = tree_thresholds[tree_thresholds != -2]
    
    # 构建完整边界
    boundaries = sorted(set([x.min() - 1] + list(tree_thresholds) + [x.max() + 1]))
    return boundaries

def calculate_woe_iv(df, feature, target, bins):
    """计算WOE和IV"""
    df['bin'] = pd.cut(df[feature], bins=bins)
    
    grouped = df.groupby('bin')[target].agg(['sum', 'count'])
    grouped.columns = ['bad', 'total']
    grouped['good'] = grouped['total'] - grouped['bad']
    
    total_good = grouped['good'].sum()
    total_bad = grouped['bad'].sum()
    
    grouped['good_dist'] = grouped['good'] / total_good
    grouped['bad_dist'] = grouped['bad'] / total_bad
    
    # WOE
    grouped['woe'] = np.log(grouped['good_dist'] / grouped['bad_dist'])
    grouped['woe'] = grouped['woe'].replace([np.inf, -np.inf], 0)
    
    # IV
    grouped['iv'] = (grouped['good_dist'] - grouped['bad_dist']) * grouped['woe']
    
    iv_total = grouped['iv'].sum()
    
    return grouped[['woe', 'iv']], iv_total
```

### 6.1 S-Learner / T-Learner / X-Learner

```python
from sklearn.ensemble import GradientBoostingRegressor

def fit_s_learner(X, T, Y):
    """S-Learner：统一模型"""
    X_aug = np.column_stack([X, T])
    model = GradientBoostingRegressor(random_state=42)
    model.fit(X_aug, Y)
    
    def predict_cate(x):
        return model.predict(np.column_stack([x, np.ones(len(x))])) - \
               model.predict(np.column_stack([x, np.zeros(len(x))]))
    
    return predict_cate

def fit_t_learner(X, T, Y):
    """T-Learner：分别建模"""
    X_treat = X[T == 1]
    Y_treat = Y[T == 1]
    X_ctrl = X[T == 0]
    Y_ctrl = Y[T == 0]
    
    model_treat = GradientBoostingRegressor(random_state=42)
    model_ctrl = GradientBoostingRegressor(random_state=42)
    
    model_treat.fit(X_treat, Y_treat)
    model_ctrl.fit(X_ctrl, Y_ctrl)
    
    def predict_cate(x):
        return model_treat.predict(x) - model_ctrl.predict(x)
    
    return predict_cate

def fit_x_learner(X, T, Y):
    """X-Learner：交叉借用"""
    # 第一阶段：T-Learner估计结果
    X_treat = X[T == 1]
    Y_treat = Y[T == 1]
    X_ctrl = X[T == 0]
    Y_ctrl = Y[T == 0]
    
    model_treat = GradientBoostingRegressor(random_state=42)
    model_ctrl = GradientBoostingRegressor(random_state=42)
    model_treat.fit(X_treat, Y_treat)
    model_ctrl.fit(X_ctrl, Y_ctrl)
    
    # 第二阶段：估计伪处理效应
    D1 = Y_treat - model_ctrl.predict(X_treat)  # 处理组伪效应
    D0 = model_treat.predict(X_ctrl) - Y_ctrl    # 对照组伪效应
    
    # 第三阶段：分别建模伪效应
    model_D1 = GradientBoostingRegressor(random_state=42)
    model_D0 = GradientBoostingRegressor(random_state=42)
    model_D1.fit(X_treat, D1)
    model_D0.fit(X_ctrl, D0)
    
    # 倾向得分（用于加权）
    from sklearn.linear_model import LogisticRegression
    ps_model = LogisticRegression(random_state=42)
    ps_model.fit(X, T)
    
    def predict_cate(x):
        g = ps_model.predict_proba(x)[:, 1]
        return g * model_D0.predict(x) + (1 - g) * model_D1.predict(x)
    
    return predict_cate
```

```python
def generate_ecommerce_data(n=10000):
    """项目01：电商大促因果推断数据"""
    np.random.seed(42)
    
    # 用户画像
    age = np.clip(np.random.normal(35, 10, n).astype(int), 18, 65)
    gender = np.random.choice([0, 1], n, p=[0.45, 0.55])
    city_tier = np.random.choice([1, 2, 3, 4], n, p=[0.15, 0.35, 0.35, 0.15])
    membership = np.random.choice([0, 1, 2, 3], n, p=[0.3, 0.4, 0.2, 0.1])
    
    # 历史行为
    historical_gmv = np.random.lognormal(5, 1.2, n)
    activity_score = np.random.beta(2, 5, n) * 100
    coupon_sensitivity = np.random.beta(2, 3, n)
    
    # 处理分配（选择偏差）
    treatment_prob = 0.1 + 0.4 * coupon_sensitivity + 0.3 * (historical_gmv / 1000)
    treatment_prob = np.clip(treatment_prob, 0.05, 0.95)
    treatment = np.random.binomial(1, treatment_prob)
    
    # 结果变量（异质性效应）
    base_gmv = 50 + 0.1 * historical_gmv + 0.5 * activity_score + 10 * membership
    treatment_effect = 30 + 50 * coupon_sensitivity
    treatment_effect *= (1 - 0.3 * (historical_gmv > np.percentile(historical_gmv, 80)))
    gmv = base_gmv + treatment * treatment_effect + np.random.normal(0, 20, n)
    
    return pd.DataFrame({
        'user_id': range(10001, 10001 + n),
        'age': age, 'gender': gender, 'city_tier': city_tier,
        'membership_level': membership,
        'historical_gmv': np.round(historical_gmv, 2),
        'activity_score': np.round(activity_score, 2),
        'coupon_sensitivity': np.round(coupon_sensitivity, 3),
        'treatment': treatment,
        'gmv': np.maximum(np.round(gmv, 2), 0)
    })
```
